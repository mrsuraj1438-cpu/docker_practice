cmd:  docker run -d --name my-sql-demo -e MYSQL_ROOT_PASSWORD=test@123 mysql : this command is used to run a container with set password 

cmd: docker exec -it my-sql-demo /bin/bash : go to the inside the my-sql-demo container 
     then -- cmd: mysql -u root -p 


mysql port : 3306
